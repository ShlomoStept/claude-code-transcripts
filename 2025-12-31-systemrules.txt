     New function render_json_with_markdown():
     def render_json_with_markdown(obj, indent=0):
         """Render a JSON object/dict with string values as Markdown."""
         if isinstance(obj, dict):
             lines = ["{"]
             items = list(obj.items())
             for i, (key, value) in enumerate(items):
                 comma = "," if i < len(items) - 1 else ""
                 rendered_value = render_json_with_markdown(value, indent + 
     2)
                 lines.append(f'{"  " * (indent + 1)}"{key}": 
     {rendered_value}{comma}')
             lines.append(f'{"  " * indent}}}')
             return "\n".join(lines)
         elif isinstance(obj, list):
             lines = ["["]
             for i, item in enumerate(obj):
                 comma = "," if i < len(obj) - 1 else ""
                 rendered_item = render_json_with_markdown(item, indent + 2)
                 lines.append(f'{"  " * (indent + 
     1)}{rendered_item}{comma}')
             lines.append(f'{"  " * indent}]')
             return "\n".join(lines)
         elif isinstance(obj, str):
             # Render string value as Markdown, wrap in a styled span
             md_html = render_markdown_text(obj)
             return f'<span class="json-string-value">{md_html}</span>'
         elif isinstance(obj, bool):
             return "true" if obj else "false"
         elif obj is None:
             return "null"
         else:
             return str(obj)

     def group_blocks_by_type(content_blocks):
         """Group content blocks into thinking, text, and tool sections."""
         thinking_blocks = []
         text_blocks = []
         tool_blocks = []

         for block in content_blocks:
             if not isinstance(block, dict):
                 continue
             block_type = block.get("type", "")
             if block_type == "thinking":
                 thinking_blocks.append(block)
             elif block_type == "text":
                 text_blocks.append(block)
